executive_sales_model.sql
-- Executive Sales Performance Model -- Author: Gustavo Taguchi WITH monthly_sales AS ( SELECT DATE_TRUNC('month', order_date) AS month, region, rep_id, SUM(revenue) AS total_revenue, COUNT(DISTINCT client_id) AS active_clients FROM fact_orders WHERE status = 'closed_won' GROUP BY 1, 2, 3 ), ranked AS ( SELECT *, LAG(total_revenue) OVER ( PARTITION BY region, rep_id ORDER BY month ) AS prev_revenue, RANK() OVER ( PARTITION BY month ORDER BY total_revenue DESC ) AS rank_in_period FROM monthly_sales ) SELECT month, region, total_revenue, ROUND((total_revenue - prev_revenue) / NULLIF(prev_revenue, 0) * 100, 2) AS mom_growth_pct, rank_in_period FROM ranked WHERE rank_in_period <= 10 ORDER BY month DESC, rank_in_period;
